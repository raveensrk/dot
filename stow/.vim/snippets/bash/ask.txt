echo "Do you want to ___?"
PS3='Choose an option: '
options=("Yes" "No" "Quit")
select opt in "${options[@]}"
do
    case $opt in
        "Yes")
            echo "You selected $opt"
            ;;
        "No")
            echo "You selected $opt"
            ;;
        "Quit")
            break
            ;;
        *) echo "Invalid option $REPLY";;
    esac
done
